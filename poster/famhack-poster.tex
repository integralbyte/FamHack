\newcommand{\TopLeftBracket}[3]{%
  \draw[#3,line width=0.75mm]
    ($ (#1,#2) + (0,-18) $) -- (#1,#2) -- ($ (#1,#2) + (18,0) $);%

\newcommand{\TagChip}[4]{%
  \filldraw[rounded corners=3mm, fill=posterbg, draw=posterline!16, line width=0.25mm]
    (#1,#2) rectangle ++(#3,13);
  \node[anchor=center, text=posterred] at ($ (#1,#2) + ({0.5*#3},6) $)
    {\posterbold\fontsize{7.9}{7.9}\selectfont #4};
}

\newcommand{\DataBox}[4]{%

}

\newcommand{\CenteredPanelText}[5]{%
  \node[anchor=center] at (#1,#2) {%
    \parbox[c][#4mm][c]{#3mm}{\centering #5}%
  };
}

\newcommand{\QRSlot}[3]{%
  \pgfmathsetmacro{\qrInset}{5}
  \pgfmathsetmacro{\qrCrossInset}{18}
  \pgfmathsetmacro{\qrCrossFrame}{#3 - 36}
  \pgfmathsetmacro{\qrTextX}{0.5 * #3}
  \pgfmathsetmacro{\qrTextY}{0.5 * #3}
  \pgfmathsetmacro{\qrImage}{#3 - 10}
  \filldraw[
    rounded corners=7mm,
    fill=posterbg,
    draw=posterred!70,
    line width=0.7mm
  ] (#1,#2) rectangle ++(#3,#3);
  \IfFileExists{assets/images/qr-code.png}{%
    \node[anchor=south west, inner sep=0pt] at ($ (#1,#2) + (\qrInset,\qrInset) $)
      {\includegraphics[width=\qrImage mm]{assets/images/qr-code.png}};
  }{%
  \node[align=center, text=postercream] at ($ (#1,#2) + (\qrTextX,{0.5*#3+4}) $)
    {\posterbold\fontsize{18}{18}\selectfont COMING};
  \node[align=center, text=postercream] at ($ (#1,#2) + (\qrTextX,{0.5*#3-10}) $)
    {\posterbold\fontsize{18}{18}\selectfont SOON};
}%
}
